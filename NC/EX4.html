#include < stdio.h >
#include < stdlib.h >
#include < math.h >
#define F(x) (x*x*x)-(5*x)+1
#define ESP 0.0001
int main(){
        float Xi2,Xi,Xi1,Yi2,Yi,Yi1,Xn,A,B,c;
	int i=0;
	while(1){
                printf("\nenter the value of X(i-2)=");
                scanf("\n %f",&Xi2);
                printf("\n enter the value of X(i)=");
                scanf("\n %f",&Xi);
                if(F(Xi2)>0 && F(Xi)>0){
                        printf("ENTER DIFFERNT VALUES\n\n");
                }else{
                        Xi1=(Xi2+Xi)/2;
			printf("\t\t-------------------------------\n");
			printf("\t\t Iterations \t\t x\n");
			printf("\t\t-------------------------------\n");
                        while(1){
                		Yi=F(Xi);
                		Yi1=F(Xi1);
                		Yi2=F(Xi2);
                		A=(((Yi1-Yi)*(Xi2-Xi)-(Yi2-Yi)*(Xi1-Xi))/((Xi1-Xi)*(Xi2-Xi)*(Xi1-Xi2)));
                		B=((Yi1-Yi)/(Xi1-Xi))-(A*(Xi1-Xi));
                		c=sqrt(B*B-4*A*Yi);
        			if(B<0){
                			Xn=Xi-((2*Yi)/(B-c));
        			}else{
                			Xn=Xi-((2*Yi)/(B+c));
        			}
				i++;
				printf("\t\t    %d  \t\t     %f\n",i,Xn);
 				if(fabs(Xn-Xi) < ESP)
   					goto jmp;
				 Xi=Xn;
 			}
			jmp:
 				printf("\n Root of the equation is %f \n",Xn=floor(Xn*10000)/10000);
	             		exit(0);
              }
	}
	return 0;
}
