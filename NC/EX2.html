Newton Rapson Method

 

#include < stdio.h >
#include < stdlib.h >
#include < math.h >
#define F(x) (pow(x,3)-(2*x)+5)
#define dF(x)(3*pow(x,2)-2)
int main(){
	float a,b,x0,x1;
	int c=0;
	while(1){
		printf("enter the value of a=");
		scanf("\n %f",&a);
		printf("\n enter the value of b=");
		scanf("\n %f",&b);
		if(F(a)>0 && F(b)>0){
			printf("ENTER DIFFERNT VALUES\n");
		}else{
			x0=(a+b)/2;
			printf("\n x0=%f \t F(x0)=%f \t dF(x0)=%f \n",x0,F(x0),dF(x0));
			printf("\t------------------------------------------------------------------");
			printf("\n \t   Iterations \t\t x \t\t F(x) \t\t dF(x)");
			printf("\n\t------------------------------------------------------------------");
			while(1){
				x1=x0-(F(x0)/dF(x0));
				c++;
				if(x0==x1){
					break;
				}
				x0=x1;
				printf("\n\t\t%d \t   %f \t    %f \t     %f",c,x0,F(x0),dF(x0));
			}
			printf("\n\n The root of equation is %f\n",x0);
			exit(0);
		}
	}
	return 0;
}
