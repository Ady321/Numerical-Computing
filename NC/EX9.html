Trapezoidal Method

 

#include < stdio.h >
#include < stdlib.h >
#include < math.h >
#define F(x) 1/(1+(x*x))
int main(){
	float a,b,h=0.25,res,y=0,integral;
	int i,N;
	printf("enter the value of upper limit and lower limit \n");
	scanf("%f\n%f",&b,&a);
	N=(b-a)/h;
	for(i=1;i < N;i++){
		res=F(i*h);
		y=y+res;
	}
	integral=(h/2)*(F(a)+F(b)+2*y);
	printf("the value of the integral is %.4f\n",integral);
}
