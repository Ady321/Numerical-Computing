#include < stdio.h >
#include < stdlib.h >
#include < math.h >
#define ESP 0.0001
#define F(x) (x*x*x*x)-(3*x*x*x)+(20*x*x)+(44*x)+54
int main(){
  	double a[10]={1,-3,20,44,54},b[10],c[10],p,q,p1,q1,p2,q2,m,dP,dQ;
  	int i=1,k,n=4;
  	printf("\nEnter the value of p: ");
  	scanf("%lf",&p);
  	printf("\nEnter the value of q: ");
  	scanf("%lf",&q);
 	printf("\t\t p \t\t q\n");
  	b[-1]=c[-1]=0;
  	b[0]=a[0];
  	c[0]=a[0];
	for(k=1;k<=4;k++){
		b[k]=a[k]-p*b[k-1]-q*b[k-2];
		c[k]=b[k]-p*c[k-1]-q*c[k-2];
	}
  	m=c[n-2]*c[n-2]-c[n-3]*(c[n-1]-b[n-1]);
  	dP=-((b[n]*c[n-3]-b[n-1]*c[n-2])/m);
  	dQ=-((b[n-1]*(c[n-1]-b[n-1])-b[n]*c[n-2])/m);
  	p1=p+dP;
  	q1=q+dQ;
  	printf("\n\t     %lf \t     %lf",p1,q1);
  	do{
		p=p1;
		q=q1;
	 	for(k=1;k<=4;k++){
                	b[k]=a[k]-p*b[k-1]-q*b[k-2];
                	c[k]=b[k]-p*c[k-1]-q*c[k-2];
        	}
  		m=c[n-2]*c[n-2]-c[n-3]*(c[n-1]-b[n-1]);
  		dP=-((b[n]*c[n-3]-b[n-1]*c[n-2])/m);
  		dQ=-((b[n-1]*(c[n-1]-b[n-1])-b[n]*c[n-2])/m);
  		p2=p+dP;
  		q2=q+dQ;
  		printf("\n\t     %lf \t     %lf",p2,q2);
		if(fabs(p1-p2) < ESP && fabs(q1-q2) < ESP){
           		printf("\n\nThe qudratic factor of the given polynomial is \n\n\tx^2 + (%.4lf)x + (%.4lf)\n",p2,q2);
    			i=0;
   		}else{
     			p1=p2;
     			q1=q2;
   		}
  	}while(i!=0);
	return 0;
}
