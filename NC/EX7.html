#include < stdio.h >
#include < stdlib.h >
#include < math.h >
#define F(y,z) ((17-y+2*z)/20)
#define G(x,z) ((-18-3*x+z)/20)
#define H(x,y) ((25-2*x+3*y)/20)
int main(){
  	double x=0,y=0,z=0,x1,y1,z1;
  	int i=0,k=1;
  	printf("\n\t Iterations        x\t\t     y\t\t     z\n");
  	printf("\n\t       %d        %.4f  \t   %.4f   \t   %.4f",k,x,y,z);
  	do{
   		x1=F(y,z);
   		y1=G(x,z);
   		z1=H(x,y);
		k++;
   		if(fabs(x1-x)<0.0001 && fabs(y1-y)<0.0001 && fabs(z1-z)<0.0001){
			printf("\n\nfinal Answer is :\n");
     			printf("\nx1=%.4lf \nx2=%.4lf \nx3=%.4lf\n",x1,y1,z1);
     			i=1;
   		}else{
     			x=x1;
     			y=y1;
     			z=z1;
     			printf("\n\t       %d        %.4f  \t  %.4f   \t   %.4f",k,x,y,z);
   		}
  	}while(i!=1);
	return 0;
}
