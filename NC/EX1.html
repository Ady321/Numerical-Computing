False Position Method

 

#include < stdio.h >
#include < stdlib.h >
#include < math.h >
#define ESP 0.0001
#define F(x) (x*x*x*x)-(x)-10
int main(){
		float a,b,x,f1,f2,f0;
  		int count=0;
  		do{
  			printf("\nEnter the value of a: ");
  			scanf("%f",&a);
  		}while(F(a) > 0);
  		do{
  			printf("\nEnter the value of b: ");
 			scanf("%f",&b);
  		}while(F(b) < 0);
  		printf("\n    a\t       b\t x\t   f0\t   f1\t   f2");

  		do{
  			f0=F(a);
  			f1=F(b);
  			x=((a*f1)-(b*f0))/(f1-f0);
  			f2=F(x);
  			printf("\n%f %f %f %f %f %f",a,b,x,f0,f1,f2);

  			if(f0*f2<0){ 
    				b=x;
   			}else{
    				a=x;
			}
  		}while(fabs(f2)>ESP);
		printf("\n\n Root of the equation using false position method is = %f\n",x);
}
