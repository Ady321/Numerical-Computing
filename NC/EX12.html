Runge Kutta Method

 

#include < stdio.h >
#include < stdlib.h >
#include < math.h >
#define F(x,y) (x)+(y)
int main(){
       float x0,y0,h,n,k1,k2,k3,k4,Dy,y;
        printf("enter the value x0,y0,width & last point\n");
        scanf("%f\n%f\n%f\n%f",&x0,&y0,&h,&n);
        for(x0=0;x0 < n;x0=x0+h){
                k1 = h*(F(x0,y0));
		k2 = h*(F(x0+h/2,y0+k1/2));
		k3 = h*(F(x0+h/2,y0+k2/2));
		k4 = h*(F(x0+h,y0+k3))
		Dy = (k1+2*k2+2*k3+k4)/6;
		y = y0+Dy;
		printf("%f\t%f\t%f\t%f\t%f\t\n",k1,k2,k3,k4,y);
		y0=y;
	}
}
