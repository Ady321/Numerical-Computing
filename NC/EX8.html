#include < stdio.h >
#include < stdlib.h >
#include < math.h >
int main(){
	float arr[3][3]={{4,1,0},{1,20,1},{0,1,4}};
	int i=0,j,sum=0;
	float x[3]={1,1,1};
	float x1[3]={0,0,0};
	float y=1,y1=0,gr;
	do{
		y=y1;
		for(i=0;3>i;i++){
			for(j=0;3>j;j++){
				sum=sum+arr[i][j]*x[j];
			}
			x1[i]=sum;
			sum=0;
		}
		gr=x1[0];
                for(i=1;i<=2;i++){
                	if(gr < x1[i]){
                         	gr=x1[i];
                 	}
                }
                y1=gr;
                for(i=0;i<=2;i++){
                	x[i]=x1[i]/(gr*1.0);
                }
	}while(y!=y1);
		printf("largest eigen value is %.4f\n",y);
		printf("corresponding eigen vector is \n");
		for(i=0;i<=2;i++){
			printf("%.4f\n",x[i]);
		}
}
