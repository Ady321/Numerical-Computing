#include < stdio.h >
#include < stdlib.h >
#include < math.h >
#define F(x) (1/(1+x*x))
int main(){
        int a,i;
        float b,n,h,res,y=0,intg;
        printf("enter the value of upper limit and lower limit\n");
        scanf("%f\n%d",&b,&a);
        printf("enter the value of the width\n");
        scanf("%f",&h);
        n=(b-a)/h;
        for(i=1;i < n;i++){
                if(i%3!=0){
                        res=3*F(i*h);
                        y=y+res;
                }else{
                        res=2*F(i*h);
                        y=y+res;
                }
        }
        intg=(3*h/8)*(F(a)+F(b)+y);
        printf("\n%.4f",intg);
}
